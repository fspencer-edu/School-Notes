**Stripes Vertex Shader**

_Vertex Shader for Drawing Stripes_

```c
uniform vec3 LightColor;
uniform vec3 EyePosition;
uniform vec3 Specular;
uniform vec3 Ambient;

uniform float Kd;
uniform mat4 MVMatrix;
uniform mat4 MVPMatrix;
uniform mat3 NormalMatrix;

in vec3 MCNormal;
in vec2 TexCoord0;

out vec3 DiffuseColor;
out vec3 SpecularColor;
out float TexCoord;


void main()
{
	vec3 ecPosition = vec3(MVMatrix * MCVertex);
	vec3 tnorm = normalize(NormalMatrix * MCNormal);
	vec3 lightVec = normalize(LightPosition - ecPosition);
	vec3 viewVec = normalize(EyePosition - ecPosition);
	vec3 hvec = normalize(viewVec + lightVec);

	float spec = clamp(dot(hvec, tnorm), 0.0, 1.0);
	spec = pow(spec, 16.0);

	DiffuseColor = LightColor * vec(Kd * dot(lightVec, tnorm));
	DiffuseColor = clamp(Ambient + DiffuseColor, 0.0, 1.0);
	SpecularColor = clamp((LightColor * Specular * spec) 0.0, 1.0);
```

```c
uniform vec4 HalfSpace[5]; // half-spaces used to define star pattern
uniform float StripeWidth;
uniform float InOrOutInit; // -3.0
uniform float FWidth; // = 0.005

uniform vec4 StarColor;
uniform vec4 StripeColor;
uniform vec4 BaseColor;

uniform vec4 LightDir; // light direction, should be normalized
uniform vec4 HVector; // reflection vector for infinite light
uniform vec4 SpecularColor;
uniform float SpecularExponent;

uniform float Ka;
uniform float Kd;
uniform float Ks;

in vec3 OCPosition; // surface position in object coordinates
flat in vec4 ECBallCenter; // ball center in eye coordinates

{
	vec3 normal; // Analytically computed normal
	vec4 pShade; // Point in shader space

	vec4 distance; // Computed distance values

	pShade.w = 1.0;

	distance[1] = dot(pShade, HalfSpace[1]);
	distance[2] = dot(pShade, HalfSpace[2]);
```

# Vertex Shader

- Shader is responsible for computing surface-local direction to the light and the surface-local direction to the eye

_Vertex Shader for Doing Procedural Bump Mapping_
```c
#version 330 core

uniform vec3 LightPosition;

uniform mat4 MVMatrix;
uniform mat4 MVPMatrix;
uniform mat3 NormalMatrix;

in vec4 MCVertex;
in vec3 MCNormal
in vec3 MCTangent;
in vec2 TexCoord0;

out vec3 LightDir;
out vec3 EyeDir;
out vec2 TexCoord;

void main()
{
	EyeDir = vec3(MVMatrix *MCVertex);
	TexCoord = TexCoord0.st;
	vec3 n = normalize(NormalMatrix * MCNormal);
	vec3 t = normalize(NormalMatrix * MCTangent);
	vec3 b = cross(n, t);
	vec3 v;
	v.x = dot(LightPosition, t);
	v.x = dot(LightPosition, b);
	v.x = dot(LightPosition, n);
	LightDir = normalize(v);
	v.x = dot(EyeDir, t);
	v.x = dot(EyeDir, b);
	v.x = dot(EyeDir, n);
	EyeDir = normalize(v);
	gl_Position = MVPMatrix * MCVertex;
}
```

# Fragment Shader

- Multiply incoming texture coordinate by the density value

_Fragment Shader for Procedural Bump Mapping_
```c
#version 330 core

uniform vec4 SurfaceColor;
uniform float BumpDensity;
uniform float BumpSize;
uniform float SpecularFactor;

in vec3 LightDir;
in vec3 EyeDir;
in vec2 TexCoord;

out vec4 FragColor;

void main()
{
	vec3 litColor;
	vec2 c = BumpDensity * TexCoord.st;
	vec2 p = fract(c) - vec2(0.5);

	float d, g;
	d = dot(p, p);
	f = inversesqrt(d + 1.0);

	if (d >= BumpSize){
		p = vec2(0.0);
		f = 1.0;
	}

	vec3 normDelta = vec3(p.x, p.y, 1.0) * f;
	litColor = SurfaceColor.rgb * max(dot(normDelta, LightDir), 0.0);
	vec reflectDir = reflect(LightDir, normDelta);

	float spec = max(dot(EyeDir, reflectDir), 0.0);
	spec = pow(spec, 6.0);
	spec *= SpecularFactor;
	litColor = min(litColor + spec, vec3(1.0));
	FragColor = vec4(litColor, SurfaceColor.a);
}
```

# Anti-aliasing Procedural Textures

## Sources of Aliasing

- Aliasing can be explained by sampling theory, while specific forms of aliasing can be explained by specific situations

- From a sampling theory perspective, a graphics image is made from samples of that scene
- If patterns vary at a high spatial frequency, the sampled can't accurately reproduce
- A period pattern needs to be sampled at least twice when is has a pattern changing faster than every 2 samples
	- Moire patterns in static image
	- Sparkling in moving image
- Impossible to correctly sample an edge with point sampled without undersampling

- A computer display is made up of a finite number of discrete elements (pixels)
- Each pixel can produce only a single colour
- Aliasing occurs when we reproduce a signal with an insufficient sampling frequency
- Eliminate aliasing by reducing the spatial frequency in the image to half the spatial frequency of the pixels
	- Loss of detail
	- Blurriness
	- Noise

![[Pasted image 20250924140957.png]]

- Point sampling is the simplest way to fetch a value from a texture when mapping it onto a surface
- Temporal aliasing is caused by objects moving too rapidly for the time sampling frequency being used
	- Stutter
	- Blink

## Avoiding Aliasing

- Design a shader that looks good while rendering an object at a specific size
- Aliasing occurs when rendering an object at different sizes
- Mipmap textures address this 
- Design a shader for each size
- Using texture instead of computing something procedurally
- Linear filtering between adjacent texels is only a solution to aliasing when the resolution of the texels is similar to the pixels

# Increasing Resolution

- Effects of aliasing can be reduced through brute force method called supersampling that performs sampling at several locations within a pixel and averages the results of those samples
- Replaces a single-point sampling operation with a several-point sampling operation
- Uses hardware resource
- High-frequency components can be misrepresented

![[Pasted image 20250924141646.png]]

- Supersampling can also be implemented within a fragment shader
- Results are improved if the sample positions are varied stochastically rather than spaced on a regular grid

```c
uniform vec3 Color1;
uniform vec3 AvgColor;
uniform float Frequency;

{

				(1.0 - smoothstep(vec2(0.0), fuzz, checkPos));

	else

		color = AvgColor;
	}
```

# Definition of Noise

- The idea noise function for modeling, rendering, or animation
	- Not show any obvious regular or repeated patterns
	- Continuous function, and its derivative is also continuous
	- Function is repeatable across time
	- Well-defined range of output values
	- Isotropic (same in all direction)
	- Defined in 1-4+ dimensions
	- Fast to compute

```c
	int frequency = startFrequency;
	GLubyte *ptr;
	double amp = 0.5;

	if ((noise3DTexPtr = (GLubyte *) malloc(noise3DTexSize *
											noise3DTexSize *
									noise3DTexSize * 4)) == NULL )
	{
		fprintg(stderr, "Error");
		exit(1);
	}
	for (f = 0; inc = 0; f < numOctaves: ++f, frequency *= 2, ++inc, 
		amp *= 0.5)
	{
		setNoiseFrequency(frequency);
		ptr = noise3DTexPtr;
		ni[0] = ni[1] = ni[2] = 0;

		inci = 1.0 / (noise3DTexSize / frequency);
		for (i = 0; j < noise3DTexSize; ++i, ni[0] += inci)
		{
			incj = 1.0 / (noise3DTexSize / frequency);
			for (j = 0; j < noise3DTexSize; ++j, ni[i] += incj)
			{
				inck = 1.0 . (noise3DTexSize / frequency);
				for (k = 0; k < noise3DTexSize;
					++k, ni[2] += inck, ptr += 4)
				{
					*(ptr +int) = (GLubyte)((noise3(ni) + 1.0) * amp) *
													 128.0);
				}
			}
		}
	}
}
```

# Trade-offs

- Use GLSL built-in noise if
	- Does not consume any texture memory
	- Does not use a texture unit
	- Continuous function rather than discrete
	- Shaders do not depend on application
- Advantages of using a texture map
	- Noise function is computed by the application
	- Store 4 noise values
	- Accessing a texture map may be faster than called the built-in noise


# A Simple Noise Shader

- Simple noise to produce a cloud effect

**Application Setup**

- Vertex position must be passed, and the surface normal
- Colours and scale factors are parameterized as uniform variables

**Vertex Shader**

- Vertex shader transforms the incoming vertex values and stores it in the built-in special variable `gl_Position`
- Vertex shader computes the light intensity
- Vertex shader scales the incoming vertex values and stored it in the out variable `MCposition`

- Fragments always produce the same position values

_Cloud Vertex Shader_
```c
#version 330 core

uniform mat4 MVMatrix;
uniform mat4 MVPMatrix;
uniform mat3 NormalMatrix;

uniform vec3 LightPos;
uniform float Scale;

in vec4 MCvertex;
in vec3 MCnormal;

out float LightIntensity
out vec3 MCposition;

void main()
{
	vec3 ECposition = vec3(MVMatrix * MCVertex);
	MCposition = vec3(MCVertex) * Scale;
	vec3 tnorm = normalize(vec3(NormalMatrix * MCNormal));
	LightIntensity = dot(nornalize(LightPos - ECposition), tnorm);
	LightIntensity *= 1.5;
	gl_Position = MVPMatrix * MCVertex;
}
```

**Fragment Shader**

- Fragment shader receives `LightIntensity and MCposition`
- These values are computed at each vertex and interpolated across the primitive by the rasterization hardware
- Compute the value of intensity by summing the 4 components of noise texture in fragment shader

_Fragment Shader for Cloudy Sky Effect_
```c
#version 330 core

uniform sampler3D Noise;
uniform vec3 SkyColor;
uniform vec3 CloudColor;

in float LightIntensity;
in vec3 MCposition;

out vec4 FragColor;

void main()
{
	vec4 noisevec = texture(Noise, MCposition);
	float intensity = (noisevec[0] + noisevec[1] +
						noisevec[2] + noisevec[3] + 0.03125) * 1.5;

	vec3 color = mix(SkyColor, CloudColor, intensity) * LightIntensity;
	FragColor = vec4(color, 1.0);
}
```

# Wood

- Wood is composed of light and dark areas alternating in concentric cylinders surrounding a central axis
- Noise is added to warp
- The centre is taken to be the y axis
- A high-frequency grain pattern gives the appearance of sawed wood


**Application Setup**

```c

Scale 2.0
LightWood 0.6, 0.3, 0.1
DarkWood 0.4, 0.2, 0.07
RingFreq 4.0

DarkGrains 0.0
GrainThreshold 0.5
NoiseScale 0.5, 0.1, 0.1
Noisiness 3.0

```


**Fragment Shader**

_Fragment Shader for Wood_
```c
#version 330 core
LightPos 0.0, 0.0, 4.0
Scale 2.0
LightWood 0.6, 0.3, 0.1
DarkWood 0.4, 0.2, 0.07
RingFreq 4.0
LightGrains 1.0
DarkGrains 0.0
GrainThreshold 0.5
NoiseScale 0.5, 0.1, 0.1
Noisiness 3.0
GrainScale 27.0

in float LightIntensity;
in vec3 MCposition;

out vec4 FragColor;

void main()
{
	vec3 noisevec = vec3(texture(Noise, MCposition * NoiseScale) *
									Noiseiness);

	vec3 location = MCposition + noisevec;
	float dist = sqrt(location.x * location.x + location.z * location.z);
	dist *= RingFreq;
	float r = fract(dist + noisevec[0] + noisevec[1] + noisevec[2]) * 2.0;

	if (r > 1.0)
		r = 2.0 - r;
	vec3 color = mix(LightWood, DarkWood, r);
	r = fract((MCposition.x + MCposition.z) * GrainScale + 0.5);
	noisevec[2] *= r;

	if (r < GrainThreshold)
		color += LightWoord * LightGrains * noisevec[2];
	else
		color -= LightWood * DarkGrains * noisevec[2];
	color *= LightIntensity;
	FragColor = vec4(color, 1.0);
}
```
